# 测试

测试相关技术笔记。

```bash
#!/bin/bash

pv_ip=192.168.1.20
echo "测试节点的IP地址为：$pv_ip"
pu_ip=$(curl https://ifconfig.me/ip 2>/dev/null)
echo "获取到我的公网节点IP为：$pu_ip"

# UI端口测试
if output=$(curl -H "Authorization: Bearer lb123456" http://$pu_ip:9090 2>/dev/null); then
    echo "UI端口测试"
    echo "命令输出：$output"
fi

# http代理测试
if output=$(curl -x http://admin:passwd@$pu_ip:7890 https://ifconfig.me/ip 2>/dev/null); then
    echo "http代理测试"
    echo "命令输出：$output"
fi

# socks5代理测试
if output=$(curl --socks5 admin:passwd@$pu_ip:7891 https://ifconfig.me/ip 2>/dev/null); then
    echo "socks5代理测试"
    echo "命令输出：$output"
fi

```
