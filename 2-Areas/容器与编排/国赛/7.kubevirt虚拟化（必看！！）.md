### 【题目1】VM管理--创建VM[3.0分]
使用镜像fedora-virt:v1.0在default命名空间下创建一台vm，名称为vm-fedora，内存为1G。

完成后提交master节点的IP地址、用户名和密码到答题框。（[http://10.24.1.46/fedora-virt_v1.0.tar](http://10.24.1.46/fedora-virt_v1.0.tar)）

[root@k8s-master-node1 ~]# docker load -i fedora-virt_v1.0.tar

[root@k8s-master-node1 ~]# cat vm.yaml

```yaml
apiVersion: kubevirt.io/v1
kind: VirtualMachine
metadata:
  name: vm-fedora
  labels:
    kubevirt.io/vm: vm-fedora
spec:
  running: false
  template:
    metadata:
      labels:
        kubevirt.io/vm: vm-fedora
    spec:
      domain:
        resources:
          requests:
            memory: 1Gi
        devices:  #设备允许添加磁盘、网络接口和其他
          disks:  #磁盘描述连接到vmi
          - name: containerdisk
            disk:   #将卷作为磁盘连接到vmi
              bus: virtio  #总线指示要仿真的磁盘设备的类型,支持的值:virtio,sata,scsi
      volumes:  #属于vmi的磁盘可以装载的卷列表
      - name: containerdisk
        containerDisk:  #ContainerDisk引用docker映像，嵌入qcow或原始磁盘
          image: fedora-virt:v1.0
        # imagePullPolicy: #镜像拉取策略，默认为always
        # path: #路径定义容器中磁盘文件的路径
```

[root@k8s-master-node1 ~]# kubectl apply -f vm.yaml

virtualmachine.kubevirt.io/vm-fedora created

[root@k8s-master-node1 ~]# kubectl get vm

NAME        AGE   STATUS    READY

vm-fedora   18s   Stopped   False

### 【题目2】存储与卷--创建emptyDisk卷[1.0分]
使用镜像fedora-virt:v1.0在default命名空间下创建一台vmi，名称为vmi-fedora，并使用emptyDisk卷为vmi挂载一块2G的磁盘。

完成后提交master节点的IP地址、用户名和密码到答题框。

[root@k8s-master-node1 ~]# docker load -i fedora-virt_v1.0.tar

[root@k8s-master-node1 ~]# cat vmi.yaml

apiVersion: kubevirt.io/v1

kind: VirtualMachineInstance

metadata:

  name: vmi-fedora

  labels:

    special: vmi-fedora

spec:

  domain:

    devices:

      disks:

      - name: containerdisk

        disk:

          bus: virtio

      - name: emptydisk

        disk:

          bus: virtio

      - name: cloudinitdisk  #这一段都可以不要

        disk:

          bus: virtio

    resources:

      requests:

        memory: 1024M

  volumes:

  - name: containerdisk

    containerDisk:

      image: fedora-virt:v1.0

  - name: emptydisk

    emptyDisk:

      capacity: 2Gi

  - name: cloudinitdisk #前面那段不要，那这段也不要

    cloudInitNoCloud:

      userData: |-

        password: fedora

        chpasswd: { expire: False }

#下面这个必须要修改，不然vmi运行不起来，vm也是同样道理

[root@k8s-master-node1 ~]# kubectl edit -n kubevirt kubevirt kubevirt

spec:

  certificateRotateStrategy: {}

  configuration:

    developerConfiguration: {}   #去掉{}

      useEmulation: true   #加上这行,开启软件虚拟化

  customizeComponents: {}

  imagePullPolicy: IfNotPresent

  workloadUpdateStrategy: {}

[root@k8s-master-node1 ~]# kubectl apply -f vmi.yaml

virtualmachineinstance.kubevirt.io/vmi-fedora created

[root@k8s-master-node1 ~]# kubectl get vmi  #要几秒启动

### 【题目3】KubeVirt 运维--创建 VMI[1.5 分]

将提供的镜像 exam.qcow2 转换为 docker 镜像 exam:v1.0，然后使用镜像 exam:v1.0 镜像在 default 命名空间下创建一台 vmi，名称为 exam，将虚拟机的 80 端口以 NodePort 的方式对外暴露为 30082，并使用数据源在启动时将 VM 的主机名初始化为exam。

完成后提交 master 节点的 IP 地址、用户名和密码到答题框。

1. qcow2 镜像转化成功得 0.3 分；
1. VMI 创建成功得 0.3 分；
1. 端口暴露成功得 0.3 分；
2. 服务能正常访问得 0.4 分；
3. 主机名初始化成功得 0.2 分。

```yaml
[root@k8s-master-node1 ~]# cat Dockerfile
FROM scratch
ADD exam.qcow2 /disk/  #一定要写/disk/ 不要写成/disk
```

```yaml
[root@k8s-master-node1 ~]# docker build -t exam:v1.0 -f Dockerfile .
[root@k8s-master-node1 ~]# docker tag exam:v1.0 192.168.2.70/library/exam:v1.0
[root@k8s-master-node1 ~]# docker push 192.168.2.70/library/exam:v1.0
[root@k8s-master-node1 ~]# cat exam-vmi.yaml
apiVersion: kubevirt.io/v1
kind: VirtualMachineInstance
metadata:
  name: exam
  labels:
    special: key
spec:
  domain:
    devices:
      disks:
      - name: containerdisk
        disk:
          bus: virtio
      - name: cloudinitdisk
        cdrom:
          bus: sata
          readonly: true
    resources:
      requests:
        memory: 2G
  volumes:
  - name: containerdisk
    containerDisk:
      image: 192.168.2.70/library/exam:v1.0
  - name: cloudinitdisk
    cloudInitNoCloud:
      userData: |
        hostname: exam
---
apiVersion: v1
kind: Service
metadata:
  name: vmi-nodeport
spec:
  type: NodePort
  selector:
    special: key
  ports:
  - name: nodeport
    port: 80
    targetPort: 80
nodePort: 30082
```

[root@k8s-master-node1 ~]# kubectl get vmi #注意要等出现ip（等ip出来且，跟主机配置有关，配置越好，IP出来越快，本人亲测7分钟左右出现有效ip）

### 【题目6】KubeVirt运维--启用热迁移
在KubeVirt中启用实时迁移功能，以实现将一台正在运行的虚拟机实例从一个节点迁移到另一个节点，期间工作负载继续运行且能保持持续访问。

完成后提交master节点的用户名、密码和IP地址到答题框。

[root@k8s-master-node1 ~]# kubectl edit -n kubevirt kubevirts.kubevirt.io

spec:

  certificateRotateStrategy: {}

  configuration:

    developerConfiguration:

      useEmulation: true

      featureGates:

      - Snapshot

      - LiveMigration  # 开启热迁移

      - DataVolumes

kubectl -n kubevirt get kubevirt kubevirt -o jsonpath={.spec.configuration.developerConfiguration.featureGates}

LiveMigration

### 【题目8】KubeVirt运维--动态密钥注入
使用exam:v1.0镜像在Kubernetes集群default命名空间下创建一台名为chinaskill-vm的VM，运行策略为Always，并使用qemuGuestAgent对该VM进行动态密钥注入，具体要求如下：

（1）基于master节点的id_rsa.pub创建一个名为my-pub-key的Secret；

（2）使用qemuGuestAgent在VM上引用这个Secret。

完成后创建该VM，并提交master节点的用户名、密码和IP地址到答题框。

```yaml
apiVersion: kubevirt.io/v1
kind: VirtualMachine
metadata:
  name: chinaskill-vm
spec:
  runStrategy: Always
  template:
    spec:
      domain:
        devices:
          disks:
          - disk:
              bus: virtio
            name: qemu-guest-agent
        resources:
          requests:
            memory: 1024M
      terminationGracePeriodSeconds: 0 #可以不写

          source:
            secret:
              secretName: my-pub-key
          propagationMethod:
            qemuGuestAgent:
              users:  #Users表示应该拥有ssh公钥的来宾用户列表
              - fedora
      volumes:
      - containerDisk:
          image: exam:v1.0
        name: qemu-guest-agent
```
