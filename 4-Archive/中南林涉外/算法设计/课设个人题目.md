## 亲密数（因子和）
```bash
def generate_divisors_sum(limit):
    """
    预生成每个数字的因子和。

    对于每个数字 i，将其加到所有大于 i 的倍数的位置。
    例如：对于 i = 2，所有 2 的倍数 (4, 6, 8, ...) 的因子和将加上 2。

    参数:
        limit (int): 生成因子和的范围上限。

    返回:
        list: 包含 0 到 limit 的每个数字的因子和。
    """
    # 初始化因子和表，divisors_sum[i] 表示数字 i 的所有真因子之和
    divisors_sum = [0] * (limit + 1)

    # 遍历每个数字 i，i 是所有大于 i 的倍数的因子
    for i in range(1, limit + 1):
        # 将 i 加到所有 i 的倍数的因子和中
        for j in range(2 * i, limit + 1, i):  # 从 2*i 开始，避免加上自己
            divisors_sum[j] += i

    return divisors_sum

def find_amicable_numbers(limit):
    """
    使用预生成的因子和表，找出小于等于 limit 的所有亲密数对。

    参数:
        limit (int): 查找亲密数对的范围上限。

    输出:
        打印出所有亲密数对。
    """
    # 获取所有数字的因子和表
    divisors_sum = generate_divisors_sum(limit)
    print(divisors_sum)

    # 遍历范围内的所有数字，查找亲密数对
    for a in range(2, limit + 1):  # 从 2 开始，因为 1 没有亲密数
        b = divisors_sum[a]  # 获取数字 a 的因子和
        # 检查 b 是否是有效的亲密数条件：
        # 1. b > a，确保不重复输出相同的数对
        # 2. b <= limit，确保 b 在范围内
        # 3. divisors_sum[b] == a，确认 b 的因子和是 a
        if b > a and b <= limit and divisors_sum[b] == a:
            # 打印亲密数对
            print(f"亲密数对: {a} 和 {b}")

if __name__ == "__main__":
    """
    主程序：设置范围并调用函数，找出范围内的所有亲密数对。
    """
    # 设置查找范围
    limit = 10000

    # 输出查找结果
    print(f"找出 {limit} 以内的所有亲密数对:")
    find_amicable_numbers(limit)

```
