# 云硬盘管理

通过华为云 Python SDK（EVS）管理云硬盘，支持创建、查询和删除操作。

## 云硬盘类型对照

| 参数 | 极速型SSD V2（公测） | 极速型SSD | 通用型SSD V2 | 超高IO | 通用型SSD | 高IO |
| :--- | :--- | :--- | :--- | :--- | :--- | :--- |
| API名称 | ESSD2 | ESSD | GPSSD2 | SSD | GPSSD | SAS |

## 国赛硬盘创建
```python
# create_block_store.py
# coding: utf-8
import time

from huaweicloudsdkcore.auth.credentials import BasicCredentials
from huaweicloudsdkevs.v2.region.evs_region import EvsRegion
from huaweicloudsdkcore.exceptions import exceptions
from huaweicloudsdkevs.v2 import *

if __name__ == "__main__":

    ak = "759QBVMFVXHSECCJJISD"
    sk = "iGEOwBehhQUdpgVtutht6HjxigJl7hBJWLxAC8sq"

    credentials = BasicCredentials(ak, sk) \

    client = EvsClient.new_builder() \
        .with_credentials(credentials) \
        .with_region(EvsRegion.value_of("cn-east-3")) \
        .build()

    try:
        # 查询同名卷的id号
        request = ListVolumesRequest()
        request.name = "chinaskills_volume"
        response = client.list_volumes(request)
        print("查询所有云硬盘：",response, "\n")
        for i in response.volumes:
            name = i.to_dict()["name"]
            id = i.to_dict()["id"]
            # 删除同名卷
            if name == "chinaskills_volume":
                request = DeleteVolumeRequest()
                request.volume_id = id
                response = client.delete_volume(request)
                print("找到第一个chinaskills_volume-id:", id)
                print("删除成功", response, "\n")
                time.sleep(1)

        # 创建云硬盘
        request = CreateVolumeRequest()
        # 启动加密
        listMetadataVolume = {
            "__system__encrypted": "1"
        }
        volumebody = CreateVolumeOption(
            availability_zone="cn-east-3a",
            multiattach=True,
            name="chinaskills_volume",
            size=10,
            volume_type="SSD",
            metadata=listMetadataVolume
        )
        request.body = CreateVolumeRequestBody(
            volume=volumebody
        )
        response = client.create_volume(request)
        print("创建成功：",response, "\n")

        # 查询创建的硬盘的id号   #response.volume_ids是一个列表使用下标获取id值
        volume_id = response.volume_ids[0]
        time.sleep(10) #等待10秒等云盘状态变为可用

        # 查询详细参数
        request = ShowVolumeRequest()
        request.volume_id = volume_id
        response = client.show_volume(request)
        print("输出详情：",response)

    except exceptions.ClientRequestException as e:
        print(e.status_code)
        print(e.request_id)
        print(e.error_code)
        print(e.error_msg)

```
