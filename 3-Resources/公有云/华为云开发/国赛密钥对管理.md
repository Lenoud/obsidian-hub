# 国赛密钥对管理

通过华为云 Python SDK 创建、查询和删除 SSH 密钥对，密钥对名称为 chinaskills_keypair。

## 代码

```python
from huaweicloudsdkcore.auth.credentials import BasicCredentials
from huaweicloudsdkecs.v2.region.ecs_region import EcsRegion
from huaweicloudsdkcore.exceptions import exceptions
from huaweicloudsdkecs.v2 import *

if __name__ == "__main__":
    ak="759QBVMFVXHSECCJJISD"
    sk="iGEOwBehhQUdpgVtutht6HjxigJl7hBJWLxAC8sq"

    credentials = BasicCredentials(ak, sk) \

    client = EcsClient.new_builder() \
        .with_credentials(credentials) \
        .with_region(EcsRegion.value_of("cn-east-3")) \
        .build()

    try:
        # 查找密钥对
        request = NovaShowKeypairRequest()
        request.keypair_name = "chinaskills_keypair"
        response = client.nova_show_keypair(request)
        name = response.to_dict()
        print(name)

        # 删除同名密钥对
        request = NovaDeleteKeypairRequest()
        request.keypair_name = "chinaskills_keypair"
        response = client.nova_delete_keypair(request)
        print(response)
    except exceptions.ClientRequestException as e:
        print(e.status_code)
        print(e.request_id)
        print(e.error_code)
        print(e.error_msg)

    try:
        # 创建密钥对
        request = NovaCreateKeypairRequest()
        keypairbody = NovaCreateKeypairOption(
            name="chinaskills_keypair"
        )
        request.body = NovaCreateKeypairRequestBody(
            keypair=keypairbody
        )
        response = client.nova_create_keypair(request)
        print(response)
    except exceptions.ClientRequestException as e:
        print(e.status_code)
        print(e.request_id)
        print(e.error_code)
        print(e.error_msg)
```
