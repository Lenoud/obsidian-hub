# 部署ERP管理系统
【适用平台】私有云

## 【题目1】容器化部署MariaDB
编写Dockerfile文件构建mysql镜像，要求基于centos完成MariaDB数据库的安装与配置，并设置服务开机自启。

root@k8s-master-node1 ~]# tar -zxvf ERP.tar.gz

[root@k8s-master-node1 ~]# cd ERP/

[root@k8s-master-node1 ERP]# docker load -i CentOS_7.9.2009.tar

[root@k8s-master-node1 ERP]# cat local.repo

[yum]

name=yum

baseurl=file:///opt/yum

gpgcheck=0

enabled=1

[root@k8s-master-node1 ERP]# cat Dockerfile-mariadb

FROM centos:centos7.9.2009

RUN rm -rf /etc/yum.repos.d/*

ADD yum/ /opt/yum

COPY local.repo /etc/yum.repos.d/

RUN yum install -y mariadb mariadb-server

COPY jsh_erp.sql /opt/

COPY init.sh /opt/

ENV LC_ALL en_US.UTF-8

RUN chmod a+x /opt/init.sh

RUN /opt/init.sh

EXPOSE 3306

CMD ["mysqld_safe","--user=root"]

[root@k8s-master-node1 ERP]# cat init.sh

#!/bin/bash

mysql_install_db --user=root

mysqld_safe --user=root &

sleep 8

mysqladmin -uroot password '123456'

mysql -uroot -p123456 -e "grant all on *.* to 'root'@'%' identified by '123456'; flush privileges;"

mysql -uroot -p123456 -e "create database jsh_erp; use jsh_erp; source /opt/jsh_erp.sql;"

[root@k8s-master-node1 ERP]# docker build -t erp-mysql:v1.0 -f Dockerfile-mariadb .

## 【题目2】容器化部署Redis
编写Dockerfile文件构建redis镜像，要求基于centos完成Redis服务的安装和配置，并设置服务开机自启。

[root@k8s-master-node1 ERP]# cat Dockerfile-redis

FROM centos:centos7.9.2009

RUN rm -rf /etc/yum.repos.d/*

COPY yum/ /opt/yum

COPY local.repo /etc/yum.repos.d/

RUN yum install -y redis

RUN sed -i 's/127.0.0.1/0.0.0.0/g' /etc/redis.conf && \

    sed -i 's/protected-mode yes/protected-mode no/g' /etc/redis.conf

RUN echo "requirepass 123456" >> /etc/redis.conf

EXPOSE 6379

CMD ["/usr/bin/redis-server","/etc/redis.conf"]

[root@k8s-master-node1 ERP]# docker build -t erp-redis:v1.0 -f Dockerfile-redis .

## 【题目3】容器化部署Nginx
编写Dockerfile文件构建nginx镜像，要求基于centos完成Nginx服务的安装和配置，并设置服务开机自启。

[root@k8s-master-node1 ERP]# cat Dockerfile-nginx

## 【题目4】容器化部署ERP
编写Dockerfile文件构建erp镜像，要求基于centos完成JDK环境和ERP服务的安装与配置，并设置服务开机自启。

[root@k8s-master-node1 ERP]# cat Dockerfile-erp

FROM centos:centos7.9.2009

RUN rm -rf /etc/yum.repos.d/*

COPY yum/ /opt/yum

COPY local.repo /etc/yum.repos.d/

RUN yum install -y java-1.8.0-openjdk java-1.8.0-openjdk-devel

RUN yum -y install mariadb-server

COPY app.jar /jshERP-boot/

COPY start.sh /opt/

RUN chmod 777 /opt/start.sh

EXPOSE 9999

CMD ["/bin/bash","/opt/start.sh"]

[root@k8s-master-node1 ERP]# docker build -t erp-server:v1.0 -f Dockerfile-erp .

## 【题目5】编排部署ERP管理系统
编写docker-compose.yaml文件，要求使用镜像mysql、redis、nginx和erp完成ERP管理系统的编排部署。

[root@k8s-master-node1 ERP]# cat docker-compose.yaml

version: '3'

services:

  erp-redis:

    container_name: erp-redis

    image: erp-redis:v1.0

    ports:

    - 6379:6379

    restart: always

  erp-mysql:

    container_name: erp-mysql

    image: erp-mysql:v1.0

    ports:

    - 3306:3306

    restart: always

  erp-server:

    container_name: erp-server

    image: erp-server:v1.0

    hostname: erp-server

    environment:

      DB_USERNAME: 123456  #坑点，jar包故意和下面那条写反了

      DB_PASSSWORD: root  #坑点，jar包故意和上面那条写反了

      REDIS_HOST: erp-redis

      REDIS_PASSWORD: 123456

    ports:

    - 9999:9999

    links:

    - erp-redis

    - erp-mysql

    restart: always

  erp-nginx:

    container_name: erp-nginx

    image: erp-nginx:v1.0

    ports:

    - 8080:80

    links:

    - erp-server:erp-server

    restart: always

[root@k8s-master-node1 ERP]# docker-compose up -d

浏览器输入http://IP:8080访问华夏erp系统（两个账号，分别是admin和jsh，密码都是123456）
