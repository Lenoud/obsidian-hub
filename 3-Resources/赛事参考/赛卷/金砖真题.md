#### 2. yum源配置(2分)
在 controller 节点安装 ftp 服务, 使用提供的 CentOS 镜像和 OpenStack 软件包配置 yum 源, 使用该 ftp 源作为 compute 节点安装 OpenStack 平台的网络源. 分别设置 controller 节点和 compute 节点的 yum 源文件为 local.repo , ftp.repo .

在 compute 节点将 cat /etc/yum.repos.d/ftp.repo 命令的返回结果提交到答题框.

**operating:**

```bash
# controller
# 删除原有的软件源
rm -rf /etc/yum.repos.d/*

# 写入本地源
cat <<EOF > /etc/yum.repos.d/local.repo
[centos]
name=centos
baseurl=file:///opt/centos/
gpgcheck=0

[iaas]
name=iaas
baseurl=file:///opt/iaas/openstack/
gpgcheck=0
EOF

# 清除缓存并安装相关软件
yum clean all
yum install vsftpd -y

# 将 /opt 目录公开
echo anon_root=/opt >> /etc/vsftpd/vsftpd.conf

# 关闭防火墙和 SELinux
systemctl disable --now firewalld
setenforce 0

# 启动 vsftpd
systemctl enable --now vsftpd
```

```bash
# compute
# 删除原有的软件源
rm -rf /etc/yum.repos.d/*

# 写入本地源
cat <<EOF > /etc/yum.repos.d/ftp.repo
[centos]
name=centos
baseurl=ftp://controller/centos/
gpgcheck=0

[iaas]
name=iaas
baseurl=ftp://controller/iaas/openstack/
gpgcheck=0
EOF

# 关闭防火墙和 SELinux
systemctl disable --now firewalld
setenforce 0
```

**answer:**

```bash

pipeline {
    agent any

    stages {
        stage('clone project') {
            steps {

            }
        }
        stage('build project') {
            steps {
                sh 'source /etc/profile; mvn package -DskipTests'
            }
        }
        stage('build images') {
            steps {
                sh 'cd config; docker build -t 192.168.100.2/springcloud/piggymetrics-config:$BUILD_ID .'
                sh 'cd gateway; docker build -t 192.168.100.2/springcloud/piggymetrics-gateway:$BUILD_ID .'
            }
        }
        stage('upload images') {
            steps {
                sh '''
                docker login 192.168.100.2 -u=admin -p=Harbor12345
                docker push 192.168.100.2/springcloud/piggymetrics-config:$BUILD_ID
                docker push 192.168.100.2/springcloud/piggymetrics-gateway:$BUILD_ID
                '''
            }
        }
        stage('deployment project') {
            steps {
                sh 'sed -i "s/sqshq\\/piggymetrics-config/192.168.100.2\\/springcloud\\/piggymetrics-config:$BUILD_ID/g" yaml/deployment/config-deployment.yaml'
                sh 'sed -i "s/sqshq\\/piggymetrics-gateway/192.168.100.2\\/springcloud\\/piggymetrics-gateway:$BUILD_ID/g" yaml/deployment/gateway-deployment.yaml'

                sh 'kubectl create ns springcloud --dry-run=client -oyaml | kubectl apply -f -'

                sh 'kubectl apply -f yaml/deployment/config-deployment.yaml ‐‐kubeconfig=/root/.kube/config'

            }
        }
    }
}
```

将 kubectl get destinationrule reviews -o jsonpath={$.spec.subsets} 命令的返回结果提交至答题框。

**operating:**

```yaml
# master
# 删除验证配置
kubectl delete validatingwebhookconfiguration istiod-default-validator

apiVersion: networking.istio.io/v1alpha3
kind: DestinationRule

  name: productpage
spec:
  host: productpage
  subsets:
  - name: v1
    labels:
      version: v1
---
apiVersion: networking.istio.io/v1alpha3
kind: DestinationRule
metadata:
  name: reviews
spec:
  host: reviews
  subsets:
  - name: v1
    labels:
      version: v1
  - name: v2
    labels:
      version: v2
  - name: v3
    labels:
      version: v3
---
apiVersion: networking.istio.io/v1alpha3
kind: DestinationRule
metadata:
  name: ratings

  host: ratings
  subsets:
  - name: v1
    labels:
      version: v1
  - name: v2
    labels:
      version: v2

apiVersion: networking.istio.io/v1alpha3

  name: details

  host: details

  - name: v1
```
